Подключение стилей

			
		<head>
			<meta charset="utf-8" />
			<title>Подключение стилей</title>

			<!--Подключаем внешние стили-->
			<link type="text/css" rel="stylesheet" href="styles.css" / >

			<!--Допоснительный слиль для другого экрана-->
			<link
				rel="stylesheet"
				type="text/css"
				href="styles480.css"
				media="screen and (max-device-width: 480px)"
			/>

			<!--Стили для печати-->
			<link
			rel="stylesheet"
			type="text/css"
			href="styles_print.css"
			media="print"
			/>

			<!--Подключение стилей в самом HTML. Атрибут type="text/css" говорит браузеру какой тип стиля будет использоваться (пока известен лишь этот тип стиля). Можно писать стили прямо здесь.-->
			<style type="text/css">
				body {
					background-color: rgb(168, 170, 168);
				}
			</style>
		</head>
